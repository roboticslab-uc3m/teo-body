248  template<typename... Ts, typename Fn>
249  void registerHandler(Fn && fn)
250  {
251  static_assert(sizeof...(Ts) > 0 && size<Ts...>() <= 8, "Illegal cumulative size.");
252  callback = [this, fn](const std::uint8_t * raw, unsigned int len)
253  { unsigned int count = 0;
254  return size<Ts...>() == len && (ordered_call{fn, unpack<Ts>(raw, &count)...}, true); };
255  }
256 
258  void unregisterHandler()
259  { callback = HandlerFn(); }

268  // https://stackoverflow.com/a/14058638
269  struct ordered_call
270  {
271  template<typename Fn, typename... Ts>
272  ordered_call(Fn && fn, Ts &&... ts)
273  { std::forward<Fn>(fn)(std::forward<Ts>(ts)...); }
274  };
